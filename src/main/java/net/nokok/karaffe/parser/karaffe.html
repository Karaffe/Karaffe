| ProtectedModifier
| PrivateModifier
| AbstractModifier
| StaticModifier
| VarModifier
| FinalModifier
| SealedModifier
| OverrideModifier
ParenFormalParams ::= "(" ( FormalParameters )? ")"
ReturnType ::= Type
FuncBody ::= "{" BlockStmts "}" 
//**********************************************************************
//Algebraic Data Type
//**********************************************************************
AlgebraicDataTypeDecl ::= Annotations ClassModifiers "case" "type" Identifier ( TypeParameters )? ( Interfaces )? "=" AlgebraicDataTypeBody ( "|" AlgebraicDataTypeBody )*
AlgebraicDataTypeBody ::= Type ( TypeParameters )? ( AlgebraicDataCtorBinding ( "," AlgebraicDataCtorBinding )* )?
AlgebraicDataCtorBinding ::= Identifier Type ( TypeParameters )? 
//**********************************************************************
//Interface
//**********************************************************************
InterfaceDecl ::= Annotations InterfaceModifiers "interface" Identifier ( TypeParameters )? ( ExtendsInterfaces )? InterfaceBody
InterfaceModifiers ::= ( InterfaceModifier )*
InterfaceModifier ::= PublicModifier
ExtendsInterfaces ::= "<-" ( InterfaceType )+
InterfaceBody ::= "{" ( InterfaceMember )* "}"
InterfaceMember ::= ( ConstantDecl | InterfaceMethodDecl | ClassDecl | InterfaceDecl )
ConstantDecl ::= ( StaticModifier )? "def" Identifier Identifier ( TypeParameters )? ( "=" VariableInitializer )?
InterfaceMethodDecl ::= ( FuncModifier )* "def" Identifier ( TypeParameters )? ParenFormalParams ( ReturnType )? ( "=" FuncBody )? 
//**********************************************************************
//TypeAlias
//**********************************************************************
TypeAliasDecl ::= Annotations TypeAliasModifiers "typealias" Identifier Identifier
TypeAliasModifiers ::= ( TypeAliasModifier )*
TypeAliasModifier ::= ( PublicModifier | ProtectedModifier | PrivateModifier ) 
//**********************************************************************
//Enum
//**********************************************************************
EnumDecl ::= Annotations EnumModifiers "enum" EnumName "[" EnumMembers "]"
EnumModifiers ::= ( EnumModifier )*
EnumModifier ::= ( PublicModifier | ProtectedModifier | PrivateModifier | StaticModifier )
EnumName ::= Identifier
EnumMembers ::= ( Identifier )+ 
//7_expr.jjt
//**********************************************************************
//Expressions
//**********************************************************************
Expression ::= ( AssignmentExpr | PipelineExpr | IfExpr | SwitchExpr | WhileLoopExpr )
AssignmentExpr ::= ( LeftHandSide AssignmentOp Expression )
AssignmentOp ::= ( "=" | "+=" | "-=" | "*=" | "/=" | "%=" | "&=" | "|=" | "^=" )
LeftHandSide ::= ExpressionName
PipelineExpr ::= BoolOrExpr ( ( "|>" | "<|" ) BoolOrExpr )*
BoolOrExpr ::= BoolAndExpr ( "|" BoolAndExpr )*
BoolAndExpr ::= EqualityExpr ( "&" EqualityExpr )*
EqualityExpr ::= ( RelationalExpr ( ( "==" | "!=" ) RelationalExpr )* )
RelationalExpr ::= ( RangeExpr ( ( "<" | ">" | "<=" | ">=" ) RangeExpr )* )
RangeExpr ::= ( AdditiveExpr ( ( ".." | ".<" ) AdditiveExpr )* )
AdditiveExpr ::= ( MultiplicativeExpr ( ( "+" | "-" | "+?" | "-?" ) MultiplicativeExpr )* )
MultiplicativeExpr ::= ( UnaryExpr ( ( "*" | "/" | "%" | "*?" | "/?" | "%?" ) UnaryExpr )* )
UnaryExpr ::= ( "+" UnaryExpr | "-" UnaryExpr | UnaryExprNotPlusMinus )
UnaryExprNotPlusMinus ::= ( "!" UnaryExpr | CastExpr | CheckExpr | PostFixExpr )
CastExpr ::= ( "(" Expression ")" "~>" ClassType )
CheckExpr ::= ( "(" Expression ")" "~>?" ClassType )
PostFixExpr ::= ( Primary | ExpressionName )
Primary ::= ( MethodInvocation | Literal | This | Self | "(" Expression ")" | ListDisplay | DictionaryDisplay | TupleDisplay | AmbiguousName )
ListDisplay ::= "[" ( Expression ( Expression )* )? "]"
DictionaryDisplay ::= "[" ( DictionaryPair )+ "]"
DictionaryPair ::= Identifier "->" Expression
TupleDisplay ::= "(" ( Expression ( Expression )* )? ")"
IfExpr ::= "if" Expression IfBody ( "else" IfBody )?
IfBody ::= ( BlockStmt | "{" BlockStmts "}" )
SwitchExpr ::= "switch" Expression "{" ( SwitchCase )+ "}"
SwitchCase ::= "case" SwitchCond "=>" CaseBody
SwitchCond ::= ( IdentifierTypeBinding | Expression ) ( SwitchGuard )?
SwitchGuard ::= "@" Expression
CaseBody ::= ( BlockStmt | "{" BlockStmts "}" )
WhileLoopExpr ::= "while" Expression WhileBody
WhileBody ::= ( BlockStmt | "{" BlockStmts "}" ) 
//**********************************************************************
//MethodInvocation
//**********************************************************************
MethodInvocation ::= ( MethodName ParenArguments ( DotMethodNameArguments )* | ExpressionName ParenArguments ( DotMethodNameArguments )* | FunctionLiteral ParenArguments ( DotMethodNameArguments )* | Literal ( DotMethodNameArguments )+ | This ( DotMethodNameArguments )+ | Self ( DotMethodNameArguments )+ | "(" Expression ")" ( DotMethodNameArguments )+ | ListDisplay ( DotMethodNameArguments )+ | DictionaryDisplay ( DotMethodNameArguments )+ | TupleDisplay ( DotMethodNameArguments )+ | AmbiguousName ( DotMethodNameArguments )+ )
ParenArguments ::= "(" ( Arguments )? ")"
DotMethodNameArguments ::= "." MethodName "(" ( Arguments )? ")"
MethodName ::= Identifier
ExpressionName ::= AmbiguousName
Arguments ::= Argument ( ( "," )? Argument )*
Argument ::= Expression
FunctionLiteral ::= FuncLiteralArg "->" FuncLiteralBody
FuncLiteralArg ::= ( "(" ( Identifier )* ")" | "(" IdentifierTypeBindings ")" | IdentifierTypeBinding | Identifier )
FuncLiteralBody ::= "{" BlockStmts "}"
IdentifierTypeBindings ::= ( IdentifierTypeBinding ( "," IdentifierTypeBinding )* )?
IdentifierTypeBinding ::= Identifier Type
AmbiguousName ::= Identifier ( "." Identifier )*

| FloatLiteral
| BoolLiteral

HexadecimalFPLiteral ::= <HexadecimalFPLiteral>
BoolLiteral ::= <BoolLiteral>
CharLiteral ::= <CharLiteral>
StringLiteral ::= <StringLiteral>
UndefinedLiteral ::= <UndefinedLiteral>
NullLiteral ::= <NullLiteral>
