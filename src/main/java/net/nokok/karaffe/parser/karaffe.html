//文法定義

/**

    型宣言

    type 識別子
    type 識別子
    type 識別子 : 継承元型名
    type 識別子              <- 実装インターフェース
    type 識別子 : 継承元型名 <- 実装インターフェース
    type 識別子 : 継承元型名[型パラメータ]
    type 識別子                            <- 実装インターフェース[型パラメータ]
    type 識別子 : 継承元型名[型パラメータ] <- 実装インターフェース[型パラメータ]
 */
TypeDeclaration ::= <Type> TypeElement ( <Colon> SuperType )? ( <ReverseArrow> Interfaces )? ( <NewLine> TypeDeclBody )?
TypeDeclBody ::= TypeDeclBody1 ( <NewLine> TypeDeclBody1 )*
TypeDeclBody1 ::= ( VariableOrFunctionDeclaration | Assignment | MethodInvocation )
SuperType ::= TypeElement
Interfaces ::= ( TypeElement )+ 
/**
    インターフェース宣言
    interface 識別子               <- 拡張インターフェース
    interface 識別子[型パラメータ] <- 拡張インターフェース[型パラメータ]
 */
InterfaceDeclaration ::= <Interface> TypeElement ( <ReverseArrow> Interfaces )? 
/**
    TypeAlias宣言
    修飾子 typealias 識別子 : 識別子

    ↓未実装
    修飾子 typealias 識別子 : 識別子[型パラメータ]
 */
TypeAliasDeclaration ::= <TypeAlias> TypeElement <Colon> TypeElement 
/**
    変数または関数の宣言

    変数宣言
    識別子 : 型名
    識別子        = 式
    識別子 : 型名
    識別子 : 型名 = 式
    識別子 : 型名[型パラメータ] = 式

    関数宣言
    識別子                      = 関数リテラル
    識別子 : 型名リスト -> 型名
    識別子 : 型名リスト -> 型名 = 関数リテラル

    識別子[型パラメータ] : 型名リスト[型パラメータ] -> 型名[型パラメータ]
    識別子[型パラメータ] : 型名リスト[型パラメータ] -> 型名[型パラメータ] = 関数リテラル
*/
VariableOrFunctionDeclaration ::= "def" Identifier ( <Colon> ValueType )? ( ( FunctionDeclBody ) | ( <EqualSign> Body ) )
FunctionDeclBody ::= <NewLine> Pattern ( <NewLine> Pattern )*
ValueType ::= TypeElement
Body ::= Expression
Pattern ::= FunctionName ArgVariables ( <Atmark> Guard )? <EqualSign> Expression
FunctionName ::= Identifier
ArgVariables ::= ( Identifier )*
Guard ::= Expression 
/**
    Enumの宣言

    enum 識別子: [識別子 識別子...]
 */
EnumDeclaration ::= <Enum> Identifier <Colon> <LeftBracket> EnumElements <RightBracket>
EnumElements ::= ( Identifier )+ 
/**
    型
    または
    型[型パラメータ]
 */
TypeElement ::= TypeWithTypeParams
| FunctionType
| <LeftParen> <RightParen>
TypeWithTypeParams ::= Identifier ( TypeParameters )?
FunctionType ::= <LeftParen> ( TypeElement )+ SingleArrow TypeElement <RightParen> 
/**
    関数リテラル

    () -> ...
    () -> { ... }
    (引数) -> ...
    (引数:型) -> ...
    (引数1 引数2...) -> ...
    (引数1:型 引数2:型 ...) -> ...
 */
FunctionLiteral ::= <LeftParen> ( Identifier )* <RightParen> SingleArrow ( <LeftBrace> Expression <RightBrace> | Expression ) 
/**
    Import文

    import 識別子
    import 識別子 -> 別名

 */
ImportStatement ::= <Import> ( ( ( Identifier | JavaFQCN ) ) ( <SingleArrow> Identifier )? )+
JavaFQCN ::= StringLiteral 
/**
    extends HogeType += hoge
*/
ExtendsType ::= "extends" TypeElement <AA> Identifier 
/**
    型パラメータ
    [型パラメータリスト]

    型パラメータリスト
    型パラメータ 型パラメータ ...

    型パラメータ

 */
TypeParameters ::= <LeftBracket> ( TypeParameter )+ <RightBracket> 
/**
    型パラメータ

    識別子
    識別子 境界型
 */
TypeParameter ::= TypeElement ( TypeBound )? 
/**
    境界型

    < 型変数
    < インターフェース & インターフェース ...
    < 型パラメータ
 */
TypeBound ::= <Lt> TypeElement ( <And> TypeElement )* 
/**
    module モジュール名
 */
ModuleDeclarationStatement ::= <Module> Identifier ModuleBody
ModuleBody ::= <NewLine> ( TypeDeclaration | VariableOrFunctionDeclaration ) ( <NewLine> ( TypeDeclaration | VariableOrFunctionDeclaration ) )* 
/**
    アノテーション

    @識別子
 */
Annotation ::= <Atmark> Identifier
Literal ::= FunctionLiteral
| BoolLiteral
| IntLiteral
| FloatLiteral
| StringLiteral
| TupleLiteral
| ListLiteral
| UndefinedLiteral
Identifier ::= <Identifier>
CompileUnit ::= ( Assignment | MethodInvocation | Statement | NewLineToken )* <EOF>
Statement ::= Declaration
| ImportStatement
| ModifierStatement
| ExtendsType
Declaration ::= VariableOrFunctionDeclaration
| TypeDeclaration
| InterfaceDeclaration
| TypeAliasDeclaration
| ModuleDeclarationStatement
| EnumDeclaration
MethodInvocation ::= ( MethodName Arguments ( ( <Dot> )? MethodName Arguments )* | FunctionLiteral Arguments )
MethodName ::= ( ElementAccess | Identifier )
ExprName ::= Identifier
Arguments ::= ( <LeftParen> <RightParen> | <LeftParen> Argument ( "," Argument )* <RightParen> )
Argument ::= ( MethodInvocation | Expression )
Expression ::= OrExpr
| Assignment
Assignment ::= LeftHandSide AssignmentOperator Expression
LeftHandSide ::= ( ElementAccess | Identifier )
AssignmentOperator ::= Assign
| AA
| SA
| MA
| DA
| RA
OrExpr ::= ( AndExpr ( Or AndExpr )* )
AndExpr ::= ( Equality ( And Equality )* )
Equality ::= ( Relational ( ( EqualTo | NotEqualTo ) Relational )* )
Relational ::= ( Range ( ( LessThan | GreaterThan | LessThanEqualTo | GreaterThanEqualTo ) Range )* )
Range ::= ( Additive ( ( ClosedRange | HalfOpenRange ) Additive )* )
Additive ::= ( Multiplicative ( ( Plus | Minus ) Multiplicative )* )
Multiplicative ::= ( UnaryExpr ( ( Star | Slash | SafeDiv | SafeRem | Percent | Hat ) UnaryExpr )* )
UnaryExpr ::= UnaryPlus UnaryExprNotPlusMinus
| UnaryMinus UnaryExprNotPlusMinus
| UnaryExprNotPlusMinus
UnaryExprNotPlusMinus ::= ( UnaryBang )* Primary
Primary ::= ( Literal | ElementAccess | MethodInvocation | Identifier | <LeftParen> Expression <RightParen> ) 
/**
    Identifier.Identifier...
 */
ElementAccess ::= ( ModuleName <DoubleColon> )? Identifier ( <Dot> Identifier )+
ModuleName ::= Identifier
ModifierStatement ::= "#" ( AllModifier )+
AllModifier ::= AbstractModifier
| OverrideModifier
| PrivateModifier
| SealedModifier
| LazyModifier
| OpModifier
| UnaryOpModifier
| VariableModifier
AbstractModifier ::= <Abstract>
OverrideModifier ::= <Override>
PrivateModifier ::= <Private>
SealedModifier ::= <Sealed>
LazyModifier ::= <Lazy>
OpModifier ::= <Operator>
UnaryOpModifier ::= <UnaryOperator>
VariableModifier ::= <Variable>
UnaryPlus ::= <Plus>
UnaryBang ::= <Bang>
UnaryMinus ::= <Minus>
NewLineToken ::= <NewLine>
Star ::= <Star>
Slash ::= <Slash>
Percent ::= <Percent>
SafeDiv ::= <SafeDiv>
SafeRem ::= <SafeRem>
Hat ::= <Hat>
Plus ::= <Plus>
Minus ::= <Minus>
EqualTo ::= <EqualTo>
NotEqualTo ::= <NotEqualTo>
GreaterThan ::= <Gt>
LessThan ::= <Lt>
GreaterThanEqualTo ::= <GtE>
LessThanEqualTo ::= <LtE>
Comparable ::= <Comparable>
NonComparable ::= <NonComparable>
And ::= <And>
Or ::= <Or>
BoolLiteral ::= <BoolLiteral>
IntLiteral ::= ( <IntLiteral> | <Hexadecimal> )
FloatLiteral ::= <FloatLiteral>
StringLiteral ::= <StringLiteral>
ListLiteral ::= <LeftBracket> ( Literal )* <RightBracket>
TupleLiteral ::= "#(" ( Expression )* <RightParen>
UndefinedLiteral ::= <Undefined>
SingleArrow ::= ( <SingleArrow> | <SingleArrowChar> )
Assign ::= <EqualSign>
ClosedRange ::= <ClosedRange>
HalfOpenRange ::= <HalfOpenRange>
AA ::= <AA>
SA ::= <SA>
MA ::= <MA>
DA ::= <DA>
RA ::= <RA>
