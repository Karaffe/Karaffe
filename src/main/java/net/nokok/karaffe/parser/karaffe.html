//文法定義

/**

    型宣言

           type 識別子
    修飾子 type 識別子
    修飾子 type 識別子 : 継承元型名
    修飾子 type 識別子              <- 実装インターフェース
    修飾子 type 識別子 : 継承元型名 <- 実装インターフェース
    修飾子 type 識別子 : 継承元型名[型パラメータ]
    修飾子 type 識別子                            <- 実装インターフェース[型パラメータ]
    修飾子 type 識別子 : 継承元型名[型パラメータ] <- 実装インターフェース[型パラメータ]
 */
TypeDeclaration ::= ( ModifierOfType )? <Type> TypeElement ( <Colon> SuperType )? ( <ReverseArrow> Interfaces )?
SuperType ::= TypeElement
Interfaces ::= ( TypeElement )+ 
/**
    インターフェース宣言
    interface 識別子               <- 拡張インターフェース
    interface 識別子[型パラメータ] <- 拡張インターフェース[型パラメータ]
 */
InterfaceDeclaration ::= <Interface> TypeElement ( <ReverseArrow> Interfaces )? 
/**
    TypeAlias宣言
    修飾子 typealias 識別子 : 識別子

    ↓未実装
    修飾子 typealias 識別子 : 識別子[型パラメータ]
 */
TypeAliasDeclaration ::= ( ModifierOfType )* <TypeAlias> TypeElement <Colon> TypeElement 
/**
    変数または関数の宣言

    変数宣言
           識別子 : 型名
    修飾子 識別子        = 式
    修飾子 識別子 : 型名
    修飾子 識別子 : 型名 = 式
    修飾子 識別子 : 型名[型パラメータ] = 式
    
    関数宣言
    修飾子 識別子                      = 関数リテラル
    修飾子 識別子 : 型名リスト -> 型名
    修飾子 識別子 : 型名リスト -> 型名 = 関数リテラル

    修飾子 識別子[型パラメータ] : 型名リスト[型パラメータ] -> 型名[型パラメータ]
    修飾子 識別子[型パラメータ] : 型名リスト[型パラメータ] -> 型名[型パラメータ] = 関数リテラル
*/
VariableOrFunctionDeclaration ::= ( Modifier )* Identifier ( ( <Colon> VariableOrFunctionTypeDeclaration <EqualSign> Expression ) | ( <Colon> VariableOrFunctionTypeDeclaration | <EqualSign> Expression ) ) 
/**
    Enumの宣言

    enum 識別子: [識別子 識別子...]
 */
EnumDeclaration ::= <Enum> Identifier <Colon> <LeftBracket> EnumElements <RightBracket>
EnumElements ::= ( Identifier )+ 
/**
    型宣言部
    
    Tは型または型[型パラメータ]

    T
    T -> T
    T -> T ... -> T
    T T ...    -> T
    T -> T T -> T
 */
VariableOrFunctionTypeDeclaration ::= TypeElement ( ( SingleArrow )? TypeElement )* 
/**
    型
    または
    型[型パラメータ]
 */
TypeElement ::= KaraffeIdentifier ( TypeParameters )? 
/**
    関数リテラル

    () -> ...
    () -> { ... }
    (引数) -> ...
    (引数:型) -> ...
    (引数1 引数2...) -> ...
    (引数1:型 引数2:型 ...) -> ...
 */
FunctionLiteral ::= <LeftParen> ( Identifier ( TypeElement )? )* <RightParen> SingleArrow ( <LeftBrace> Statements <RightBrace> | Statements ) 
/**
    Import文

    import 識別子
    import 識別子 -> 別名
    import 識別子 { }

 */
ImportStatement ::= <Import> ( Identifier ( <SingleArrow> Identifier )? )* ( <LeftBrace> Statements <RightBrace> )? 
/**
    型パラメータ
    [型パラメータリスト]

    型パラメータリスト
    型パラメータ 型パラメータ ...

    型パラメータ
    
 */
TypeParameters ::= <LeftBracket> ( TypeParameter )+ <RightBracket> 
/**
    型パラメータ

    識別子
    識別子 境界型
 */
TypeParameter ::= TypeElement ( TypeBound )? 
/**
    境界型

    < 型変数
    < インターフェース & インターフェース ...
    < 型パラメータ
 */
TypeBound ::= <Lt> TypeElement ( <And> TypeElement )* 
/**
    module モジュール名
 */
ModuleDeclarationStatement ::= <Module> Identifier 
/**
    アノテーション

    @識別子
 */
Annotation ::= <Atmark> Identifier
Literal ::= FunctionLiteral
| BoolLiteral
| IntLiteral
| FloatLiteral
| StringLiteral
| UndefinedLiteral
Identifier ::= KaraffeIdentifier
| JavaFQCN
KaraffeIdentifier ::= <Identifier>
JavaFQCN ::= <JavaFQCN>
CompileUnit ::= ( Statement | NewLineToken )* <EOF>
Statements ::= ( Statement )*
Statement ::= Declaration
| ImportStatement
Declaration ::= VariableOrFunctionDeclaration
| TypeDeclaration
| InterfaceDeclaration
| TypeAliasDeclaration
| ModuleDeclarationStatement
| EnumDeclaration
| EndOfModule 
//MethodInvocations
MethodInvocation ::= ( MethodName Arguments | TypeElement ( <Dot> )? Identifier Arguments | ExpressionName ( <Dot> )? Arguments | "" Primary ( <Dot> )? Arguments )
MethodName ::= Identifier
ExpressionName ::= Identifier
Arguments ::= ( <LeftParen> <RightParen> | ( <LeftParen> )? ( ArgumentList )? ( <RightParen> )? )
ArgumentList ::= Expression ( Expression )* 
/**
    ModuleName::ElementName...
 */
ExplicitModuleElementAccess ::= Identifier ( <DoubleColon> Identifier )+
Expression ::= ( OrExpr | AssignmentExpression )
AssignmentExpression ::= ( OrExpr | Assignment )
Assignment ::= LeftHandSide AssignmentOperator Expression
LeftHandSide ::= ( Identifier | ExplicitModuleElementAccess ( MethodInvocation )? )
AssignmentOperator ::= Assign
| AA
| SA
| MA
| DA
| RA
OrExpr ::= ( AndExpr ( Or AndExpr )* )
AndExpr ::= ( Equality ( And Equality )* )
Equality ::= ( Relational ( ( EqualTo | NotEqualTo ) Relational )* )
Relational ::= ( Range ( ( LessThan | GreaterThan | LessThanEqualTo | GreaterThanEqualTo ) Range )* )
Range ::= ( Additive ( ( ClosedRange | HalfOpenRange ) Additive )* )
Additive ::= ( Multiplicative ( ( Plus | Minus ) Multiplicative )* )
Multiplicative ::= ( UnaryExpr ( ( Star | Slash | SafeDiv | SafeRem | Percent | Hat ) UnaryExpr )* )
UnaryExpr ::= UnaryPlus UnaryExprNotPlusMinus
| UnaryMinus UnaryExprNotPlusMinus
| UnaryExprNotPlusMinus
UnaryExprNotPlusMinus ::= Primary
| UnaryBang Primary
Primary ::= Literal
| <LeftParen> Expression <RightParen>
| MethodInvocation 
/**
    Identifier=>Identifier...
 */
ElementAccess ::= Identifier ( <DoubleArrow> Identifier )+
Modifier ::= ModifierOfType
| ModifierOfVariable
| ModifierOfFunction
| ModifierOfOperator
ModifierOfType ::= ( ( PrivateModifier | SealedModifier ) )+
ModifierOfVariable ::= ( ( PrivateModifier | VariableModifier | LazyModifier ) )+
ModifierOfFunction ::= ( ( AbstractModifier | OverrideModifier | PrivateModifier | SealedModifier ) )+
ModifierOfOperator ::= ( ( OpModifier | UnaryOpModifier ) )+
AbstractModifier ::= <Abstract>
OverrideModifier ::= <Override>
PrivateModifier ::= <Private>
SealedModifier ::= <Sealed>
LazyModifier ::= <Lazy>
OpModifier ::= <Operator>
UnaryOpModifier ::= <UnaryOperator>
VariableModifier ::= <Variable>
UnaryPlus ::= <Plus>
UnaryBang ::= <Bang>
UnaryMinus ::= <Minus>
EndOfModule ::= <EndModule>
NewLineToken ::= <NewLine>
Star ::= <Star>
Slash ::= <Slash>
Percent ::= <Percent>
SafeDiv ::= <SafeDiv>
SafeRem ::= <SafeRem>
Hat ::= <Hat>
Plus ::= <Plus>
Minus ::= <Minus>
EqualTo ::= <EqualTo>
NotEqualTo ::= <NotEqualTo>
GreaterThan ::= <Gt>
LessThan ::= <Lt>
GreaterThanEqualTo ::= <GtE>
LessThanEqualTo ::= <LtE>
Comparable ::= <Comparable>
NonComparable ::= <NonComparable>
And ::= <And>
Or ::= <Or>
BoolLiteral ::= <BoolLiteral>
IntLiteral ::= ( <IntLiteral> | <Hexadecimal> )
FloatLiteral ::= <FloatLiteral>
StringLiteral ::= <StringLiteral>
UndefinedLiteral ::= <Undefined>
SingleArrow ::= ( <SingleArrow> | <SingleArrowChar> )
Assign ::= <EqualSign>
ClosedRange ::= <ClosedRange>
HalfOpenRange ::= <HalfOpenRange>
AA ::= <AA>
SA ::= <SA>
MA ::= <MA>
DA ::= <DA>
RA ::= <RA>
